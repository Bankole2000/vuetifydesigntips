// Install vue
>>>: npm install -g @vue/cli
NOTE: Select the "Vue 2" option when prompted

// create new vue app 
>>>: vue create <appname>

// Add Vuetify Plugin
>>>: vue add vuetify 
NOTE: Select the "Prototype" option when prompted

// Add Vuex Plugin
>>>: vue add vuex

// Add Router Plugin (optional)
>>>: vue add router

// Start the server 
>>>: npm run serve 

// Run deploy script 
>>>: npm run deploy
